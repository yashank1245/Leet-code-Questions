
#include <bits/stdc++.h>
using namespace std;
int main() {
  int n,i,x;
  cin>>n>>x;
  vector<int>arr(n);
  for(i=0;i<n;i++){
  	cin>>arr[i];
  }
	unordered_map<int,int>freq;

  for(int num:arr){
  	freq[num]++;
  }
  cout<<"output\n";
  for(auto entry:freq){
  	if(freq[entry.first-x]>0){
  		cout<<entry.first<<"\n";
  		cout<<entry.first-x<<"\n";
	  }
  }
  
  
  
  
}
